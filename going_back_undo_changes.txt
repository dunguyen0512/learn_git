## back to the previous version ##
git checkout # move to exact commit
git reset # reset to previous commit 
git restore # recover file to old version
git reverser # undo 
## git restore ##
git restore filename # recover file from working to the latest commit of that file. 
if the file is in staging area, we need to 2 steps.
git restore --staged filename ## move to unstage file
git restore filename ## delete changes in working directory.
git restore. ## recover the entire files.
git restore -- source head~N filename # head~N : number of commits before current commit.eg. Head~1 or Head~2
Note: git restore can not undo. 
example : 
git restore --staged .\going_back_undo_changes.txt ## move to unstage file
git restore --source HEAD~1 .\going_back_undo_changes.txt

#### git reset #####
git reset heshkey 
git reset HEAD~N

git reset has 3 options: mixed (default), --soft, and --hard.
git reset (default) # keep the current file and remove git add. 
git reset --soft # keep staged and working directory
git reset --hard # all file and changes will be deleted.
